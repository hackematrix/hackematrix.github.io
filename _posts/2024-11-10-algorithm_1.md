# 冒泡排序
* C language
```c
#include<stdio.h>
#define maxn 1010
int a[maxn];
int main(){
  int n;
  int i,j;
  int temp;
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&a[i]);
  for(i=1;i<n;i++){
    for(j=0;j<n-i;j++){
      if(a[j]>a[j+1]){
        temp=a[j];
        a[j]=a[j+1];
        a[j+1]=temp;
      }
    }
}
  for(i=0;i<n;i++)
    printf("%3d",a[i]);
  return 0;

}
